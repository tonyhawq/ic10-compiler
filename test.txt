 
number min = 0;
number max = 10;

function temp() -> number
{
	return 3;
}

function heat() -> void
{
	number to_heat = max - temp();
	asm "peek r0";
	asm "s db Setting r0";
}

function cool() -> void
{
	number to_cool = temp() - min;
	asm "peek r0";
	asm "s db Setting r0";
}

function main() -> void
{
	while (true)
	{
		number temperature = temp();
		if (temperature < min)
		{
			heat();
		}
		if (temperature > max)
		{
			cool();
		}
	}
}
